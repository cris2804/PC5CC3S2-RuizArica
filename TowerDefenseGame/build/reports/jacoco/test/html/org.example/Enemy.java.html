package org.example;

public class Enemy {
    private int speed;
    private int health;
    private int reward;
    private int x, y;

    public Enemy(int speed, int health, int reward) {
        this.speed = speed;
        this.health = health;
        this.reward = reward;
        this.x = 0;
        this.y = 2; // Starting position of enemies
    }

    public void move(int[][] grid) {
        // Movimiento simple: moverse hacia abajo si es posible, si no, hacia la derecha
        if (x + 1 < grid.length && grid[x + 1][y] == 1) {
            x++;
        } else if (y + 1 < grid[0].length && grid[x][y + 1] == 1) {
            y++;
        }
    }

    public void takeDamage(int damage) {
        this.health -= damage;
    }


    public boolean isDefeated() {
        return this.health <= 0;
    }

    public int getX() {
        return x;
    }

    public int getY() {
        return y;
    }

    public int getHealth() {
        return health;
    }

    public int getReward() {
        return reward;
    }

    public int getSpeed() {
        return speed;
    }

    public void setPosition(int x, int y) {
        this.x = x;
        this.y = y;
    }
}
